QUICK AP SANITY CHECK (run after each pair of runs)
--------------------------------------------------------------
python -c "
import json, numpy as np
from sklearn.metrics import average_precision_score, roc_auc_score
from pathlib import Path

files = [
    ('e3a Private curve', 'outputs/e3b_student_267clips_tte/tte_curve/e3a_tte_curve_private_epoch07.jsonl'),
    ('e3a Public curve',  'outputs/e3b_student_267clips_tte/tte_curve/e3a_tte_curve_public_epoch07.jsonl'),
    ('e3b Private curve', 'outputs/e3b_student_267clips_tte/tte_curve/e3b_tte_curve_private_step000099.jsonl'),
    ('e3b Public curve',  'outputs/e3b_student_267clips_tte/tte_curve/e3b_tte_curve_public_step000099.jsonl'),
    ('e3b Val',           'outputs/e3b_student_267clips_tte/metrics/e3b_val_step000099/e3b_val_step000099.jsonl'),
    ('e3b Train readback','outputs/e3b_student_267clips_tte/metrics/e3b_train_readback_step000099/e3b_train_readback_step000099.jsonl'),
]
for tag, path in files:
    if not Path(path).exists():
        print(f'{tag}: NOT READY'); continue
    rows = [json.loads(l) for l in open(path) if l.strip()]
    yt = np.array([int(r['ground_truth']) for r in rows])
    ys = np.array([float(r['score']) for r in rows])
    ap = average_precision_score(yt, ys) if len(set(yt)) > 1 else float('nan')
    print(f'{tag}: n={len(rows)}  pos={int(yt.sum())}  AP={ap:.4f}  mean_score={ys.mean():.3f}')
"

STAGE D — UPLOAD RESULTS TO HF
==============================================================

HF_HOME=/root/.cache/huggingface python -c "
from huggingface_hub import HfApi
api = HfApi()

# Upload all 6 result JSONLs to e3b-ep3-lora repo under results/tte_curve/
import glob
files = (
    glob.glob('outputs/e3b_student_267clips_tte/tte_curve/*.jsonl') +
    glob.glob('outputs/e3b_student_267clips_tte/metrics/e3b_val_step000099/*.jsonl') +
    glob.glob('outputs/e3b_student_267clips_tte/metrics/e3b_train_readback_step000099/*.jsonl')
)
for f in files:
    from pathlib import Path
    p = Path(f)
    path_in_repo = f'results/tte_curve/{p.name}'
    api.upload_file(
        path_or_fileobj=f,
        path_in_repo=path_in_repo,
        repo_id='EviatarO/e3b-ep3-lora',
        repo_type='model',
    )
    print(f'Uploaded: {path_in_repo}')
print('All results on HF.')
"
