STEP 3 — Fix HuggingFace cache  [CRITICAL — prevents disk quota crash]
         HF model is 7-9 GB. It MUST live on local disk (/root),
         NOT on /workspace (network volume has a size quota).
--------------------------------------------------------------
# Always delete stale cache from /workspace first:
rm -rf /workspace/.cache

# Check where the model cache is:
ls -la /root/.cache/huggingface

# CASE A — output shows "No such file or directory" (fresh pod, model not downloaded yet):
#   Nothing to do. Model will download to /root/.cache automatically on first run.

# CASE B — output shows a real directory (drwx...):
#   Nothing to do. Model is already on local disk.

# CASE C — output shows a symlink (lrwx... -> /workspace/...):
#   rm -rf /root/.cache/huggingface && mkdir -p /root/.cache/huggingface

# If model is stuck on /workspace (e.g. after a crash), move it:
#   mkdir -p /root/.cache
#   mv /workspace/.cache/huggingface /root/.cache/huggingface

STEP 10 — Find best checkpoint after training finishes
--------------------------------------------------------------
cd /workspace/Car-Crash-Prediction-Based-MMLM

python -c "
import json
rows = [json.loads(l) for l in open('outputs/checkpoints/e3a_lora_89clips/epoch_metrics.jsonl')]
best_ap = max(rows, key=lambda r: r.get('val_ap', 0))
best_f1 = max(rows, key=lambda r: r.get('val_f1', 0))
print(f'Best val_ap : epoch={best_ap[\"epoch\"]}  val_ap={best_ap[\"val_ap\"]:.3f}  step=step_{best_ap[\"step\"]:06d}')
print(f'Best val_f1 : epoch={best_f1[\"epoch\"]}  val_f1={best_f1[\"val_f1\"]:.3f}  step=step_{best_f1[\"step\"]:06d}')
"
