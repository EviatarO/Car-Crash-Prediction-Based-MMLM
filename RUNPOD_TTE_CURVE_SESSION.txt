import json, numpy as np
from sklearn.metrics import average_precision_score
from pathlib import Path

files = [
    ('e3a Private',  'outputs/e3b_student_267clips_tte/tte_curve/e3a_tte_curve_private_epoch07.jsonl'),
    ('e3a Public',   'outputs/e3b_student_267clips_tte/tte_curve/e3a_tte_curve_public_epoch07.jsonl'),
    ('e3b Private',  'outputs/e3b_student_267clips_tte/tte_curve/e3b_tte_curve_private_step000099.jsonl'),
    ('e3b Public',   'outputs/e3b_student_267clips_tte/tte_curve/e3b_tte_curve_public_step000099.jsonl'),
    ('e3b Val',      'outputs/e3b_student_267clips_tte/metrics/e3b_val_step000099/e3b_val_step000099.jsonl'),
    ('e3b Train',    'outputs/e3b_student_267clips_tte/metrics/e3b_train_readback_step000099/e3b_train_readback_step000099.jsonl'),
]
for tag, path in files:
    if not Path(path).exists():
        print(f'{tag:15s}: not ready'); continue
    rows = [json.loads(l) for l in open(path) if l.strip()]
    yt = np.array([int(r['ground_truth']) for r in rows])
    ys = np.array([float(r['score']) for r in rows])
    ap = average_precision_score(yt,ys) if len(set(yt))>1 else float('nan')
    print(f'{tag:15s}: n={len(rows):4d}  pos={int(yt.sum()):3d}  AP={ap:.4f}  mean_score={ys.mean():.3f}')
"
