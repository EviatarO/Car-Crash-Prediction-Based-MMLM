PYTORCH_CUDA_ALLOC_CONF=expandable_segments:True \
HF_HOME=/root/.cache/huggingface nohup python student_training/scripts/train_lora.py \

    --config      student_training/configs/train_lora.yaml \
    --init_from   outputs/checkpoints/e3a-epoch7-lora \
    --num_epochs  25 \
    > train_e3b.log 2>&1 &
